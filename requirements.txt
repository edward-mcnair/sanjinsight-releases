#   NVIDIA GPU (CUDA acceleration):
#     CMAKE_ARGS="-DLLAMA_CUDA=on" pip install llama-cpp-python --force-reinstall --no-cache-dir

# The application runs without llama-cpp-python; the AI assistant will be
